// ============================================================
// Antigravity Service Orchestrator — Frontend Integration Code
// Give this file to your frontend team member.
// Works with React, React Native, or plain JavaScript.
// ============================================================

const API_BASE = "http://localhost:8000"; // change to production URL when deploying


// ─────────────────────────────────────────────────────────────
// 1. MAIN API FUNCTION — call this to submit a service request
// ─────────────────────────────────────────────────────────────

export async function submitServiceRequest({
  userId,
  query,
  urgency = "medium",
  address,
  lat,
  lng,
}) {
  const payload = {
    user_id: userId,
    raw_query: query,
    urgency: urgency,               // "low" | "medium" | "high" | "emergency"
    location: {
      address: address,
      lat: parseFloat(lat),
      lng: parseFloat(lng),
    },
  };

  const response = await fetch(`${API_BASE}/requests/`, {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify(payload),
  });

  if (!response.ok) {
    const err = await response.json().catch(() => ({}));
    throw new Error(err.detail || `Server error: HTTP ${response.status}`);
  }

  const json = await response.json();
  return json; // see parseResponse() below for how to use this
}


// ─────────────────────────────────────────────────────────────
// 2. RESPONSE PARSER — extract what you need for the UI
// ─────────────────────────────────────────────────────────────

export function parseResponse(json) {
  const { success, data, meta } = json;

  return {
    // Top-level flags
    success,                          // boolean
    status: data.status,              // "success" | "unavailable" | "error"
    message: data.message,            // string — safe to show directly in UI

    // Booking meta
    bookingId: meta.booking_id,       // "BK-1747391234" or null
    detectedService: meta.detected_intent,   // "ac_technician" etc.
    detectedLanguage: meta.detected_language, // "roman_urdu" | "en"
    urgency: meta.urgency,            // "low" | "medium" | "high" | "emergency"

    // Provider — null if no provider found
    provider: data.provider
      ? {
          id: data.provider.id,
          name: data.provider.name,
          phone: data.provider.phone,
          rating: data.provider.rating,           // float e.g. 4.7
          distanceKm: data.provider.distance_km,  // float e.g. 1.2
          experienceYears: data.provider.experience_years,
          reasoning: data.provider.reasoning,     // why this provider was selected
        }
      : null,

    // Appointment — null if no provider found
    appointment: data.appointment
      ? {
          bookingId: data.appointment.booking_id,
          timeDisplay: data.appointment.scheduled_time_display, // "10:30 AM, 17 May"
          timeISO: data.appointment.scheduled_time_iso,         // for Date parsing
          address: data.appointment.location?.address,
          costPerHour: data.appointment.cost_per_hour,          // number e.g. 1500
          currency: data.appointment.currency,                  // "PKR"
        }
      : null,

    // Next steps — ALWAYS an array of objects, never plain strings
    // type: "action" | "info" | "warning"
    nextSteps: (data.next_steps || []).map((s) => ({
      step: s.step,               // number
      title: s.title,             // string — render as heading
      description: s.description, // string — render as body text
      type: s.type,               // "action" | "info" | "warning"
      actionValue: s.action_value, // phone number / booking id / null
      actionLabel: s.action_label, // "Call Now" / "Track" / null
    })),

    // Trace — ALWAYS an array of objects, never plain strings
    trace: (data.trace || []).map((t) => ({
      stage: t.stage,     // "intent_detection" | "provider_ranking" | etc.
      message: t.message, // clean string — no emojis
      status: t.status,   // "completed" | "pending" | "failed"
    })),

    // Followup info
    followup: data.followup || null,

    // Error info — only present when status !== "success"
    error: data.error || null,
  };
}


// ─────────────────────────────────────────────────────────────
// 3. REACT HOOK — drop this into any React / React Native screen
// ─────────────────────────────────────────────────────────────

import { useState } from "react";

export function useServiceRequest() {
  const [loading, setLoading] = useState(false);
  const [result, setResult] = useState(null);   // parsed response
  const [error, setError] = useState(null);     // error string or null

  async function submit(params) {
    setLoading(true);
    setError(null);
    setResult(null);

    try {
      const raw = await submitServiceRequest(params);
      const parsed = parseResponse(raw);
      setResult(parsed);
    } catch (e) {
      setError(e.message);
    } finally {
      setLoading(false);
    }
  }

  return { submit, loading, result, error };
}
